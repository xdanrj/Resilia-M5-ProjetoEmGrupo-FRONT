CREATE TABLE produtos(id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, nome VARCHAR(60), cor VARCHAR(20), preco FLOAT(10), img VARCHAR(300) );

CREATE TABLE filiais(id INT NOT NULL PRIMARY KEY AUTO_IMPLEMENT, nome VARCHAR(60), local VARCHAR(90), cnpj VARCHAR(30) );

CREATE TABLE vendedores(id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, nome VARCHAR(45), email VARCHAR(30), cpf VARCHAR(14), meta INT );

CREATE TABLE Fornecedores(id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, nome VARCHAR(60), endereco VARCHAR(250), cnpj VARCHAR(18) );

CREATE TABLE clientes(id INT NOT NULL PRIMARY KEY AUTO_INCREMENT, nome VARCHAR(60), email VARCHAR(50), cpf VARCHAR(18), cep VARCHAR(18) );
